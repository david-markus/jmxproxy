001package com.github.mk23.jmxproxy.conf;
002
003import com.fasterxml.jackson.annotation.JsonProperty;
004import com.fasterxml.jackson.databind.annotation.JsonDeserialize;
005import com.fasterxml.jackson.databind.annotation.JsonSerialize;
006
007import io.dropwizard.jackson.JsonSnakeCase;
008
009import io.dropwizard.util.Duration;
010
011import java.util.ArrayList;
012import java.util.List;
013
014import javax.validation.Valid;
015import javax.validation.constraints.Min;
016
017/**
018 * <p>Application configuration.</p>
019 *
020 * Maintains application configuration fields. The application initializer
021 * creates and marshals the object from JSON or Yaml config file. Setters
022 * may be chained together for configuration building.
023 *
024 * <p>For example:</p>
025 *
026 * <code>
027 * AppConfig cfg = new AppConfig()
028 *     .setCleanInterval(Duration.minutes(5))
029 *     .setCacheDuration(Duration.minutes(10))
030 *     .setHistorySize(20);
031 * </code>
032 *
033 * @see <a href="http://dropwizard.github.io/dropwizard/0.9.2/dropwizard-util/apidocs/io/dropwizard/util/Duration.html">io.dropwizard.util.Duration</a>
034 *
035 * @since   2016-01-28
036 * @author  mk23
037 * @version 3.3.6
038 */
039@JsonSnakeCase
040public class AppConfig {
041    /** Default clean interval minutes. */
042    private static final int DEFAULT_CLEAN_INTERVAL = 1;
043    /** Default cache duration minutes. */
044    private static final int DEFAULT_CACHE_DURATION = 5;
045    /** Default access duration minutes. */
046    private static final int DEFAULT_ACCESS_DURATION = 30;
047    /** Default jmx connect timeout milliseconds. */
048    private static final int DEFAULT_CONNECT_TIMEOUT = 10000;
049
050    /**
051     * Configuration for how often to run the task that finds and
052     * purges stale endpoints.
053     */
054    @Valid
055    @JsonProperty
056    @JsonDeserialize(using = DurationDeserializer.class)
057    @JsonSerialize(using = DurationSerializer.class)
058    private Duration cleanInterval = Duration.minutes(DEFAULT_CLEAN_INTERVAL);
059
060    /**
061     * Configuration for how often to reconnect to cached endpoints
062     * and scrape available attributes.
063     */
064    @Valid
065    @JsonProperty
066    @JsonDeserialize(using = DurationDeserializer.class)
067    @JsonSerialize(using = DurationSerializer.class)
068    private Duration cacheDuration = Duration.minutes(DEFAULT_CACHE_DURATION);
069
070    /**
071     * Configuration for how long an endpoint goes unaccessed before
072     * the purger tasks removes it from the cache.
073     */
074    @Valid
075    @JsonProperty
076    @JsonDeserialize(using = DurationDeserializer.class)
077    @JsonSerialize(using = DurationSerializer.class)
078    private Duration accessDuration = Duration.minutes(DEFAULT_ACCESS_DURATION);
079
080    /**
081     * Configuration for how long to wait for a JMX connection to complete
082     * before declaring the endpoint as inaccessible.
083     */
084    @Valid
085    @JsonProperty
086    @JsonDeserialize(using = DurationDeserializer.class)
087    @JsonSerialize(using = DurationSerializer.class)
088    private Duration connectTimeout = Duration.minutes(DEFAULT_CONNECT_TIMEOUT);
089
090    /**
091     * Configuration for how many fetched attribute values to keep for
092     * each cached endpoint.
093     */
094    @Min(1)
095    @JsonProperty
096    private int historySize = 1;
097
098    /**
099     * Configuration for whitelisted endpoints, allowing all when this
100     * is empty.
101     */
102    @JsonProperty
103    private List<String> allowedEndpoints = new ArrayList<String>();
104
105    /**
106     * <p>Getter for cleanInterval.</p>
107     *
108     * Configuration for how often to run the task that finds and
109     * purges stale endpoints.
110     *
111     * @return Configured clean interval.
112     */
113    public final Duration getCleanInterval() {
114        return cleanInterval;
115    }
116    /**
117     * <p>Setter for cleanInterval.</p>
118     *
119     * @param cleanInterval period at which the purger tasks operates.
120     *
121     * @return Modified AppConfig for setter chaining.
122     */
123    public final AppConfig setCleanInterval(final Duration cleanInterval) {
124        this.cleanInterval = cleanInterval;
125        return this;
126    }
127
128    /**
129     * <p>Getter for cacheDuration.</p>
130     *
131     * Configuration for how often to reconnect to cached endpoints
132     * and scrape available attributes.
133     *
134     * @return Configured cache duration.
135     */
136    public final Duration getCacheDuration() {
137        return cacheDuration;
138    }
139    /**
140     * <p>Setter for cacheDuration.</p>
141     *
142     * @param cacheDuration period at which the endpoint cache
143     * refreshes.
144     *
145     * @return Modified AppConfig for setter chaining.
146     */
147    public final AppConfig setCacheDuration(final Duration cacheDuration) {
148        this.cacheDuration = cacheDuration;
149        return this;
150    }
151
152    /**
153     * <p>Getter for accessDuration.</p>
154     *
155     * Configuration for how long an endpoint goes unaccessed before
156     * the purger tasks removes it from the cache.
157     *
158     * @return Configured access duration.
159     */
160    public final Duration getAccessDuration() {
161        return accessDuration;
162    }
163    /**
164     * <p>Setter for accessDuration.</p>
165     *
166     * @param accessDuration time before an enpoint is removed
167     * from the cache.
168     *
169     * @return Modified AppConfig for setter chaining.
170     */
171    public final AppConfig setAccessDuration(final Duration accessDuration) {
172        this.accessDuration = accessDuration;
173        return this;
174    }
175
176    /**
177     * <p>Getter for connectTimeout.</p>
178     *
179     * Configuration for how long to wait for a JMX connection to complete
180     * before declaring the endpoint as inaccessible.
181     *
182     * @return Configured connect timeout.
183     */
184    public final Duration getConnectTimeout() {
185        return connectTimeout;
186    }
187    /**
188     * <p>Setter for connectTimeout.</p>
189     *
190     * @param connectTimeout time before an abandoning a new JMX connection.
191     *
192     * @return Modified AppConfig for setter chaining.
193     */
194    public final AppConfig setConnectTimeout(final Duration connectTimeout) {
195        this.connectTimeout = connectTimeout;
196        return this;
197    }
198
199    /**
200     * <p>Getter for historySize.</p>
201     *
202     * Configuration for how many fetched attribute values to keep for
203     * each cached endpoint.
204     *
205     * @return Configured history size.
206     */
207    public final int getHistorySize() {
208        return historySize;
209    }
210    /**
211     * <p>Setter for historySize.</p>
212     *
213     * @param historySize number of values to keep for all cached
214     * attributes at every requested endpoint.
215     *
216     * @return Modified AppConfig for setter chaining.
217     */
218    public final AppConfig setHistorySize(final int historySize) {
219        this.historySize = historySize;
220        return this;
221    }
222
223    /**
224     * <p>Getter for allowedEndpoints.</p>
225     *
226     * Configuration for whitelisted endpoints, allowing all when this
227     * is empty.
228     *
229     * @return Configured {@link List} containing whitelisted endpoints.
230     */
231    public final List<String> getAllowedEndpoints() {
232        return allowedEndpoints;
233    }
234    /**
235     * <p>Setter for allowedEndpoints.</p>
236     *
237     * @param allowedEndpoints {@link List} containing whitelisted endpoints.
238     *
239     * @return Modified AppConfig for setter chaining.
240     */
241    public final AppConfig setAllowedEndpoints(final List<String> allowedEndpoints) {
242        this.allowedEndpoints = allowedEndpoints;
243        return this;
244    }
245}
